SELECT * FROM Certificate as c
join assessor as a on a.id = c.assessor_id
join postcode as p on p.id = c.postcode_id
join employer as e on e.id = c.employer_id
join type as t on t.id = c.type_id
join scheme as sch on sch.id = c.scheme_id
join complexity as comp on comp.id = c.complexity_id
join environment as env on env.id = c.environment_id
join heating as h on h.id = c.heating_id
join rrn as r on r.id = c.rrn_id
join site on site.id = c.site_id
where postcode like ''
order by c.expiry
